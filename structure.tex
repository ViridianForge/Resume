%----------------------------------------------------------------------------------------
%	JOB COMMAND
%	#1 -- Start Date
%	#2 -- End Date
%	#3 -- Location
%	#4 -- Website
%	#5 -- Title
%	#6 -- Short Description
%
%----------------------------------------------------------------------------------------

\newcommand{\job}[6]{
\begin{tabbing}
\hspace{2cm} \= \kill
\textbf{#1} \> \href{#4}{#3} \\
\textbf{#2} \>\+ \textit{#5} \\
\begin{minipage}{\smallertextwidth}
\vspace{2mm}
#6
\end{minipage}
\end{tabbing}
\vspace{2mm}
}

%----------------------------------------------------------------------------------------
%	CLASS COMMAND
%	#1 -- Term(s) Taught
%	#2 -- Department and School
%	#3 -- Class Title
%	#4 -- Details
%----------------------------------------------------------------------------------------

\newcommand{\class}[4]{
\begin{tabbing}
\hspace{2cm} \= \kill
\textbf{#1} \> {#2} \\
\> \+ \textbf{#3} \\
\begin{minipage}{\smallertextwidth}
#4
\end{minipage}
\end{tabbing}
\vspace{2mm}
}

%----------------------------------------------------------------------------------------
%	PUBLICATION COMMAND
